CSS Implementation (globals.css)

/* Scanline overlay — applied via TerminalShell ::before */
.scanline::before {
  content: '';
  position: absolute;
  inset: 0;
  background: repeating-linear-gradient(
    0deg,
    transparent,
    transparent 2px,
    rgba(0, 0, 0, 0.15) 2px,
    rgba(0, 0, 0, 0.15) 4px
  );
  pointer-events: none;
  z-index: 10;
}

/* Text glow — applied to all terminal text */
.glow {
  text-shadow:
    0 0 5px rgba(0, 255, 204, 0.4),
    0 0 10px rgba(0, 255, 204, 0.2);
}

/* Subtle flicker — applied to terminal shell */
@keyframes flicker {
  0%, 100% { opacity: 1; }
  50% { opacity: 0.97; }
}
.terminal-flicker {
  animation: flicker 4s ease-in-out infinite;
}
